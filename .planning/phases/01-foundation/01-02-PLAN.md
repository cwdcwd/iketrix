```typescript
/**
 * Normalized task input from any external source.
 * All adapters must transform their source data into this shape.
 */
export interface TaskInput {
  externalId: string;
  title: string;
  description: string | null;
  externalUrl: string | null;
  labels: string[];
  assignees: string[];
  state: "open" | "closed";
  createdAt: Date;
  updatedAt: Date;
  metadata: Record<string, unknown>; // source-specific extra data
}

/**
 * Connection configuration for an input source.
 */
export interface SourceConnection {
  id: string;
  type: string; // "github", "jira", "linear", etc.
  name: string; // human-readable name, e.g., "owner/repo"
  accessToken: string;
}

/**
 * Result of a sync operation.
 */
export interface SyncResult {
  added: TaskInput[];
  updated: TaskInput[];
  closed: TaskInput[];
  errors: Array<{ externalId: string; error: string }>;
}

/**
 * Pluggable input adapter interface.
 * Implement this for each external task source (GitHub, Jira, Linear, etc.).
 */
export interface InputAdapter {
  readonly type: string;

  /**
   * Validate that the connection credentials are still valid.
   */
  validateConnection(connection: SourceConnection): Promise<boolean>;

  /**
   * Fetch all tasks from the source. Used for initial import.
   */
  fetchAll(connection: SourceConnection): Promise<TaskInput[]>;

  /**
   * Sync changes since last sync. Returns added, updated, and closed items.
   */
  sync(connection: SourceConnection, since: Date): Promise<SyncResult>;

  /**
   * Assign a task to a user in the external system (for delegation).
   * Returns true if assignment succeeded.
   */
  assignTask?(
    connection: SourceConnection,
    externalId: string,
    assignee: string
  ): Promise<boolean>;
}
```
